      #include <stdio.h>

      int main() {
         int num_of_elements = 0;
         scanf("%d", &num_of_elements);
      
         int array[num_of_elements];
   
         for( int i = 0; i < num_of_elements; i++) {
            scanf("%d", &array[i]);
         }
      
         //array has been loaded
         for(int i = 1; i < num_of_elements; i++) {
            int j = ( i - 1 );
            int key = array[i];
         
            while( array[j] > key && ( j >= 0 )) {
               array[j+1] = array[j];
               array[j] = key;
               j--;
            }
      
            for(int i = 0; i < num_of_elements; i++) {
               printf("%d ", array[i]);
            }
            printf("\n");
         }
      }
